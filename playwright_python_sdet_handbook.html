from playwright.sync_api import Page, expect

def test_orangehrm_valid_login(page: Page) -> None:
    login_url = "https://opensource-demo.orangehrmlive.com/web/index.php/auth/login"
    page.goto(login_url)

    # Assertions for title and URL
    expect(page).to_have_title(lambda title: "OrangeHRM" in title)
    assert "auth/login" in page.url

    # Locators
    username_input = page.locator('input[name="username"]')
    password_input = page.locator('input[name="password"]')
    login_button   = page.get_by_role("button", name="Login")

    username_input.fill("Admin")
    password_input.fill("admin123")
    login_button.click()

    # Dashboard check
    expect(page).to_have_url(lambda url: "dashboard" in url)
    header = page.locator("h6.oxd-text--h6")
    expect(header).to_contain_text("Dashboard")

2.5 ‚Äì Base Page (pages/base_page.py)


from playwright.sync_api import Page

class BasePage:
    def __init__(self, page: Page):
        self.page = page

    def goto(self, url: str) -> None:
        self.page.goto(url)

    def click(self, locator: str) -> None:
        self.page.locator(locator).click()

    def fill(self, locator: str, value: str) -> None:
        self.page.locator(locator).fill(value)

2.6 ‚Äì Login Page POM (pages/login_page.py)


from playwright.sync_api import Page
from .base_page import BasePage

class LoginPage(BasePage):
    def __init__(self, page: Page):
        super().__init__(page)
        self.username_input = page.locator("input[name='username']")
        self.password_input = page.locator("input[name='password']")
        self.login_button   = page.get_by_role("button", name="Login")
        self.error_message  = page.locator(".oxd-alert-content-text")
        self.login_url      = "https://opensource-demo.orangehrmlive.com/web/index.php/auth/login"

    def open(self) -> None:
        self.page.goto(self.login_url)

    def login(self, username: str, password: str) -> None:
        self.username_input.fill(username)
        self.password_input.fill(password)
        self.login_button.click()

2.7 ‚Äì Dashboard Page POM (pages/dashboard_page.py)


from playwright.sync_api import Page, expect
from .base_page import BasePage

class DashboardPage(BasePage):
    def __init__(self, page: Page):
        super().__init__(page)
        self.header = page.locator("h6.oxd-text--h6")
        self.user_dropdown = page.locator(".oxd-userdropdown-name")
        self.logout_link = page.get_by_role("menuitem", name="Logout")

    def assert_on_dashboard(self) -> None:
        expect(self.header).to_contain_text("Dashboard")

    def logout(self) -> None:
        self.user_dropdown.click()
        self.logout_link.click()

3.9 ‚Äì Interview Notes (Python + Playwright)

Q: Why use Pytest with Playwright?

A: Pytest provides powerful test discovery, fixtures, markers, and reporting. The pytest-playwright plugin adds built-in fixtures like page, browser, context so you can focus on test logic instead of setup/teardown.

Q: How do you run Playwright tests in parallel in Python?

A: Use pytest-xdist and run pytest -n 4 or pytest -n auto, which distributes tests across multiple workers.
