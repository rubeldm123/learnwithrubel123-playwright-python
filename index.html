1.2 – Playwright Config + Parallel Execution

playwright.config.js – basic with parallel workers


// @ts-check
const { defineConfig } = require('@playwright/test');

module.exports = defineConfig({
  testDir: './tests',
  timeout: 30 * 1000,
  expect: {
    timeout: 5000
  },
  fullyParallel: true,      // allow tests in a file to run in parallel if test.describe.configure
  workers: 4,               // number of parallel workers (or use 'auto')
  retries: 1,               // retry failed tests once
  reporter: [
    ['list'],
    ['html'],
    // Allure reporter will be added in Section 2
  ],
  use: {
    browserName: 'chromium',
    headless: false,
    viewport: { width: 1280, height: 720 },
    actionTimeout: 10 * 1000,
    screenshot: 'only-on-failure',
    video: 'retain-on-failure',
    trace: 'retain-on-failure'
  }
});

Run tests in parallel from CLI


# Default (uses workers from config or auto)
npx playwright test

# Override workers from CLI (e.g., 4 parallel workers)
npx playwright test --workers=4

# Run only Chromium headed in parallel
npx playwright test --browser=chromium --headed

1.5 – Assertions Cheat Sheet (expect)


const { test, expect } = require('@playwright/test');

test('Assertion examples on OrangeHRM login page', async ({ page }) => {
  await page.goto('https://opensource-demo.orangehrmlive.com/web/index.php/auth/login');

  // Title and URL
  await expect(page).toHaveTitle(/OrangeHRM/);
  await expect(page).toHaveURL(/auth\/login/);

  // Element is visible and enabled
  const usernameInput = page.locator('input[name="username"]');
  await expect(usernameInput).toBeVisible();

  const loginButton = page.getByRole('button', { name: 'Login' });
  await expect(loginButton).toBeEnabled();

  // Text and attribute assertion
  await expect(loginButton).toHaveText('Login');
  await expect(usernameInput).toHaveAttribute('name', 'username');
});

2.5 – Page Object: OrangeHRMLoginPage.js


// pages/OrangeHRMLoginPage.js
class OrangeHRMLoginPage {
  /**
   * @param {import('@playwright/test').Page} page
   */
  constructor(page) {
    this.page = page;
    this.usernameInput = page.locator('input[name="username"]');
    this.passwordInput = page.locator('input[name="password"]');
    this.loginButton   = page.getByRole('button', { name: 'Login' });
    this.errorToast    = page.locator('.oxd-alert-content-text');
  }

  async goto() {
    await this.page.goto('/web/index.php/auth/login');
  }

  async login(username, password) {
    await this.usernameInput.fill(username);
    await this.passwordInput.fill(password);
    await this.loginButton.click();
  }

  async loginValidDefault() {
    await this.login('Admin', 'admin123');
  }
}

module.exports = { OrangeHRMLoginPage };

2.6 – Page Object: OrangeHRMDashboardPage.js


// pages/OrangeHRMDashboardPage.js
class OrangeHRMDashboardPage {
  /**
   * @param {import('@playwright/test').Page} page
   */
  constructor(page) {
    this.page = page;
    this.dashboardTitle = page.locator('h6.oxd-text--h6');
    this.userDropdown   = page.locator('.oxd-userdropdown-name');
    this.logoutLink     = page.getByRole('menuitem', { name: 'Logout' });
  }

  async assertOnDashboard() {
    await this.page.waitForURL('**/dashboard/index');
    await this.dashboardTitle.waitFor();
  }

  async logout() {
    await this.userDropdown.click();
    await this.logoutLink.click();
  }
}

module.exports = { OrangeHRMDashboardPage };

2.7 – Login Test (POM, Ready for Parallel)


// tests/login.spec.js
const { test, expect } = require('@playwright/test');
const { OrangeHRMLoginPage } = require('../pages/OrangeHRMLoginPage');
const { OrangeHRMDashboardPage } = require('../pages/OrangeHRMDashboardPage');
const users = require('../test-data/users.json');

test.describe.parallel('OrangeHRM Login Tests', () => {

  test('Valid login navigates to dashboard', async ({ page }) => {
    const loginPage = new OrangeHRMLoginPage(page);
    const dashboard = new OrangeHRMDashboardPage(page);

    await loginPage.goto();
    await loginPage.login(users.validUser.username, users.validUser.password);

    await dashboard.assertOnDashboard();
    await expect(dashboard.dashboardTitle).toContainText('Dashboard');
  });

  test('Invalid login shows error message', async ({ page }) => {
    const loginPage = new OrangeHRMLoginPage(page);

    await loginPage.goto();
    await loginPage.login(users.invalidUser.username, users.invalidUser.password);

    await expect(loginPage.errorToast).toBeVisible();
  });

});

3.9 – JavaScript + Playwright Interview Highlights

Q: Why do we use async/await in Playwright?

A: Playwright commands are asynchronous and return Promises. Using async/await makes the code look synchronous and easier to read. It also ensures the step finishes before moving to the next step.

Q: Real-time example of using array methods with Playwright?

let names = [];
for (let i = 0; i < count; i++) {
